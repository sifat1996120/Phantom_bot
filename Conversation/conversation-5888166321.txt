```python
import os
import io
import asyncio # For async operations, which Telegram bots often use
from PIL import Image # For image processing

# We'll need these for the Telegram bot part
from telegram import Update
from telegram.ext import Application, CommandHandler, MessageHandler, filters, ContextTypes

# And this for Gemini!
import google.generativeai as genai

# --- Configuration (SUPER IMPORTANT! Don't share these!) ---
# You'll need to get these from Telegram (BotFather) and Google AI Studio
TELEGRAM_BOT_TOKEN = "YOUR_TELEGRAM_BOT_TOKEN" # <--- Get this from @BotFather on Telegram!
GOOGLE_API_KEY = "YOUR_GOOGLE_API_KEY"       # <--- Get this from Google AI Studio!

# Configure the Gemini API
genai.configure(api_key=GOOGLE_API_KEY)

# --- Gemini Model Setup ---
# For text-only prompts
text_model = genai.GenerativeModel('gemini-pro')
# For multi-modal prompts (text + image)
vision_model = genai.GenerativeModel('gemini-pro-vision')

# --- Bot Command Handlers ---

# This is what happens when someone types /start
async def start(update: Update, context: ContextTypes.DEFAULT_TYPE) -> None:
    user = update.effective_user
    await update.message.reply_html(
        f"Hi {user.mention_html()}! 👋 I'm your Gemini-powered bot. "
        "Send me text, or text with a photo, and I'll try to answer! "
        "Try sending a photo with a caption like 'What is this?'"
    )

# This is for handling regular text messages
async def handle_text_message(update: Update, context: ContextTypes.DEFAULT_TYPE) -> None:
    user_message = update.message.text
    print(f"Received text message from {update.effective_user.first_name}: {user_message}") # Just for debugging!

    try:
        # Send the message to Gemini and get a response
        response = text_model.generate_content(user_message)
        await update.message.reply_text(response.text)
    except Exception as e:
        print(f"Error calling Gemini API for text: {e}")
        await update.message.reply_text("Oops! Something went wrong trying to get an answer from Gemini. 🥺")

# This is the cool part for handling photos with captions!
async def handle_photo_message(update: Update, context: ContextTypes.DEFAULT_TYPE) -> None:
    # Get the photo with the highest resolution
    photo_file = await update.message.photo[-1].get_file()
    photo_bytes = io.BytesIO()
    await photo_file.download_to_memory(photo_bytes)
    photo_bytes.seek(0) # Go back to the beginning of the bytes stream

    # Open the image using Pillow
    img = Image.open(photo_bytes)

    # Get the caption (if any)
    caption = update.message.caption or "Describe this image." # Default prompt if no caption

    print(f"Received photo from {update.effective_user.first_name} with caption: {caption}") # Debugging!

    try:
        # Send both the image and text to the vision model
        response = vision_model.generate_content([caption, img])
        await update.message.reply_text(response.text)
    except Exception as e:
        print(f"Error calling Gemini API for vision: {e}")
        await update.message.reply_text("Oh no! I had trouble understanding that photo. Can you try another one? 😥")

# --- Main function to run the bot ---
handler(CommandHandler("start", start))
    application.add_handler(MessageHandler(filters.TEXT & ~filters.COMMAND, handle_text_message))
    application.add_handler(MessageHandler(filters.PHOTO & filters.CAPTION, handle_photo_message)) # For photos with captions
    application.add_handler(MessageHandler(filters.PHOTO & ~filters.CAPTION, handle_photo_message)) # For photos without captions

    # Run the bot until you press Ctrl-C
    print("Bot is running! Press Ctrl-C to stop.")

if __name__ == "__main__":
    main()
```
